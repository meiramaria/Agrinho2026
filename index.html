document.addEventListener("DOMContentLoaded", () => { // ============================ // Botões // ============================ const exploreBtn = document.getElementById("exploreBtn"); const infoBtn = document.getElementById("infoBtn"); if (exploreBtn) { exploreBtn.addEventListener("click", () => { document.getElementById("praticas").scrollIntoView({ behavior: "smooth" }); }); } if (infoBtn) { infoBtn.addEventListener("click", () => { document.getElementById("sobre").scrollIntoView({ behavior: "smooth" }); }); } // ============================ // Menu mobile // ============================ const menuToggle = document.getElementById("menuToggle"); const nav = document.querySelector("nav"); if (menuToggle && nav) { menuToggle.addEventListener("click", () => { nav.classList.toggle("open"); }); } // ============================ // Barra de progresso // ============================ window.addEventListener("scroll", () => { const progress = document.querySelector(".progress"); const docHeight = document.body.scrollHeight - window.innerHeight; const scrollTop = window.scrollY; if (progress) { const percent = (scrollTop / docHeight) * 100; progress.style.width = percent + "%"; } }); // ============================ // QUIZ (CORRIGIDO) // ============================ const quizData = [ { pergunta: "Qual prática reduz o consumo de água em até 50%?", opcoes: [ { texto: "Irrigação por gotejamento", correta: true }, { texto: "Plantio direto", correta: false }, { texto: "Uso de pesticidas", correta: false } ] }, { pergunta: "O plantio direto ajuda principalmente a:", opcoes: [ { texto: "Aumentar erosão", correta: false }, { texto: "Preservar o solo", correta: true }, { texto: "Desmatar áreas", correta: false } ] }, { pergunta: "Controle biológico usa:", opcoes: [ { texto: "Produtos químicos fortes", correta: false }, { texto: "Inimigos naturais das pragas", correta: true }, { texto: "Queima de lavouras", correta: false } ] } ]; const quizContainer = document.getElementById("quizContainer"); const submitQuiz = document.getElementById("submitQuiz"); const quizResult = document.getElementById("quiz-result"); // renderizar perguntas if (quizContainer) { quizContainer.innerHTML = ""; quizData.forEach((q, qi) => { const block = document.createElement("div"); block.classList.add("quiz-block"); const pergunta = document.createElement("p"); pergunta.classList.add("quiz-pergunta"); pergunta.textContent = q.pergunta; const opcoes = document.createElement("div"); opcoes.classList.add("quiz-opcoes"); q.opcoes.forEach((op, oi) => { const label = document.createElement("label"); label.classList.add("quiz-opcao"); const input = document.createElement("input"); input.type = "radio"; input.name = "q" + qi; input.value = oi; if (op.correta) { input.dataset.correct = "true"; } label.appendChild(input); label.appendChild(document.createTextNode(op.texto)); opcoes.appendChild(label); }); block.appendChild(pergunta); block.appendChild(opcoes); quizContainer.appendChild(block); }); } // resultado if (submitQuiz) { submitQuiz.addEventListener("click", () => { let score = 0; quizData.forEach((q, qi) => { const selected = document.querySelector(`input[name="q${qi}"]:checked`); const inputs = document.querySelectorAll(`input[name="q${qi}"]`); inputs.forEach(i => { i.disabled = true; if (i.dataset.correct === "true") { i.parentElement.classList.add("correta"); } }); if (selected) { if (selected.dataset.correct === "true") { score++; selected.parentElement.classList.add("acertou"); } else { selected.parentElement.classList.add("errou"); } } }); const percent = Math.round((score / quizData.length) * 100); quizResult.style.display = "block"; quizResult.innerHTML = `
🌱

Você acertou ${score} de ${quizData.length} (${percent}%)

Continue aprendendo sobre sustentabilidade!

`; submitQuiz.style.display = "none"; }); } });
